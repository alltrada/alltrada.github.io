    const sections = document.querySelectorAll('.section-view');


    navLinks.forEach(link => {

      link.addEventListener('click', (e) => {

        const sectionId = e.currentTarget.dataset.section + '-section';

        

        sections.forEach(section => {

          section.classList.remove('active');

        });


        navLinks.forEach(link => link.classList.remove('active'));


        const targetSection = document.getElementById(sectionId);


        if (targetSection) {

          targetSection.classList.add('active');

          e.currentTarget.classList.add('active');


          // Scroll zurücksetzen

          targetSection.scrollTop = 0;

        }

      });

    });


    if (navLinks.length > 0) {

      navLinks[0].classList.add('active');

    }


    const defaultConfig = {

      company_name: 'alltrada',

      hero_tagline: 'Trade & Distribution of Consumer Goods',

      primary_color: '#C6A75E',

      text_color: '#2B2B2B',

      background_color: '#F3F1EB',

      font_family: 'IBM Plex Sans'

    };


    let config = { ...defaultConfig };


    async function onConfigChange(newConfig) {

      config = { ...config, ...newConfig };

      

      if (config.company_name) {

        document.getElementById('nav-company').textContent = config.company_name;

      }


      if (config.hero_tagline) {

        document.getElementById('hero-tagline').textContent = config.hero_tagline;

      }


      if (config.primary_color) {

        document.querySelectorAll('.section-heading-accent').forEach(el => {

          el.style.backgroundColor = config.primary_color;

        });

      }


      if (config.text_color) {

        document.body.style.color = config.text_color;

      }


      if (config.background_color) {

        document.body.style.backgroundColor = config.background_color;

      }


      if (config.font_family) {

        document.body.style.fontFamily = config.font_family + ', sans-serif';

      }

    }


    function mapToCapabilities(cfg) {

      return {

        recolorables: [

          {

            get: () => cfg.primary_color || defaultConfig.primary_color,

            set: (value) => {

              config.primary_color = value;

              window.elementSdk.setConfig({ primary_color: value });

            }

          },

          {

            get: () => cfg.text_color || defaultConfig.text_color,

            set: (value) => {

              config.text_color = value;

              window.elementSdk.setConfig({ text_color: value });

            }

          },

          {

            get: () => cfg.background_color || defaultConfig.background_color,

            set: (value) => {

              config.background_color = value;

              window.elementSdk.setConfig({ background_color: value });

            }

          }

        ],

        borderables: [],

        fontEditable: {

          get: () => cfg.font_family || defaultConfig.font_family,

          set: (value) => {

            config.font_family = value;

            window.elementSdk.setConfig({ font_family: value });

          }

        },

        fontSizeable: undefined

      };

    }


    function mapToEditPanelValues(cfg) {

      return new Map([

        ['company_name', cfg.company_name || defaultConfig.company_name],

        ['hero_tagline', cfg.hero_tagline || defaultConfig.hero_tagline]

      ]);

    }


    if (window.elementSdk) {

      window.elementSdk.init({

        defaultConfig,

        onConfigChange,

        mapToCapabilities,

        mapToEditPanelValues

      });

    } else {

      onConfigChange(defaultConfig);

    }


    function getActiveSection() {

      return document.querySelector('.section-view.active');

    }


    function updateHeaderOffsetToSectionScrollbar() {

      const s = getActiveSection();

      if (!s) return;


      requestAnimationFrame(() => {

        const sbw = Math.max(0, s.offsetWidth - s.clientWidth);

        document.documentElement.style.setProperty('--section-sbw', sbw + 'px');

      });

    }


    window.addEventListener('load', updateHeaderOffsetToSectionScrollbar);

    window.addEventListener('resize', updateHeaderOffsetToSectionScrollbar);


    document.querySelectorAll('.nav-link').forEach(link => {

      link.addEventListener('click', () => {

        setTimeout(updateHeaderOffsetToSectionScrollbar, 0);

        setTimeout(updateHeaderOffsetToSectionScrollbar, 50);

        setTimeout(updateHeaderOffsetToSectionScrollbar, 150);

      });

    });


    // Mobile menu toggle

    const mobileBtn = document.getElementById('mobile-menu-btn');

    const mobileMenu = document.getElementById('mobile-menu');


    if (mobileBtn) {

      mobileBtn.addEventListener('click', () => {

        mobileMenu.classList.toggle('hidden');

      });

    }


    // Close mobile menu on nav click

    document.querySelectorAll('#mobile-menu .nav-link').forEach(link => {

      link.addEventListener('click', () => {

        mobileMenu.classList.add('hidden');

      });

    });

  </script>
